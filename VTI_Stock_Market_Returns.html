D2D = function (x) {
    days = nrow(x)
    delta = numeric(days)
    for(i in 2:days){
        delta[i] <- (100*((x[i, 1] - x[i - 1, 1]) / (x[i - 1, 1])))
    }
    delta
}
VTI.InterDay = D2D(z)
VTI.InterDay[1] <- mean(VTI.InterDay)#Something to fill in the 0 in row 1.
